#### Postgres ####

# Linux: Get & decode postgres password
kubectl get secret -n kexa my-postgresql -o jsonpath="{.data.postgres-password}" | base64 --decode ; echo

# Windows: Get postgres password
kubectl get secret --namespace kexa my-postgresql -o jsonpath="{.data.postgres-password}"

# Windows: Decode hashed Postgres password
[System.Text.Encoding]::UTF8.GetString([System.Convert]::FromBase64String("YOUR_HASHED_PASSWORD_HERE"))
